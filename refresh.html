<head>

    <meta charset="UTF-8">

    <meta name="viewport" content="width=device-width, initial-scale=1.0">

    <title>Website Refresher</title>

</head>

<body>

    <h1>Website Refresher</h1>

    <p>Enter a website URL and click the button to refresh it 100 times.</p>

    <input type="text" id="urlInput" placeholder="Enter website URL (e.g., https://example.com)">

    <button onclick="startRefreshing()">Start Refreshing</button>

    <p id="status">Status: Waiting to start...</p>


    <script>

        function startRefreshing() {

            const url = document.getElementById('urlInput').value;

            const status = document.getElementById('status');


            // Validate URL

            if (!url.startsWith('http://') && !url.startsWith('https://')) {

                status.textContent = 'Error: Please enter a valid URL starting with http:// or https://';

                return;

            }


            // Open the website in a new window

            const newWindow = window.open(url, '_blank');

            if (!newWindow) {

                status.textContent = 'Error: Failed to open the website. Please allow pop-ups.';

                return;

            }


            let refreshCount = 0;

            const maxRefreshes = 100;

            const refreshInterval = 1000; // 1 second delay between refreshes


            status.textContent = `Refreshing ${url}... (0/${maxRefreshes})`;


            // Function to refresh the page

            function refreshPage() {

                if (refreshCount < maxRefreshes && !newWindow.closed) {

                    newWindow.location.reload();

                    refreshCount++;

                    status.textContent = `Refreshing ${url}... (${refreshCount}/${maxRefreshes})`;

                    setTimeout(refreshPage, refreshInterval);

                } else {

                    if (!newWindow.closed) {

                        status.textContent = `Completed ${maxRefreshes} refreshes for ${url}.`;

                    } else {

                        status.textContent = 'Stopped: The website window was closed.';

                    }

                }

            }


            // Start refreshing

            setTimeout(refreshPage, refreshInterval);

        }

    </script>

</body>

</html>
